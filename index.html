Installation

# Clone the repository
git clone https://github.com/mohammad-arch-svg/shell.git

# Navigate to directory
cd shell/shell_project

# Create build directory
mkdir build && cd build

# Configure with CMake
cmake ..

# Build the project
make -j$(nproc)

# Run the shell
./shell
